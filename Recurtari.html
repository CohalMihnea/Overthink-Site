// GALERIE

const gallerySlides = document.getElementById("gallerySlides");

const prevBtn = document.querySelector(".gallery .prev");

const nextBtn = document.querySelector(".gallery .next");

let galleryIndex = 0;

let allowTransition = true;

let slideWidth = 310;


if(window.innerWidth > 768 && gallerySlides){

  gallerySlides.innerHTML += gallerySlides.innerHTML;


  function calcSlideWidth(){

    const firstImg = gallerySlides.querySelector("img");

    if(firstImg){

      const computedStyle = window.getComputedStyle(firstImg);

      const marginRight = parseFloat(computedStyle.marginRight) || 0;

      slideWidth = firstImg.getBoundingClientRect().width + marginRight;

    }

  }


  window.addEventListener("load", calcSlideWidth);

  window.addEventListener("resize", calcSlideWidth);

  setTimeout(calcSlideWidth, 200);


  function moveGallery(step){

    if(!allowTransition) return;


    galleryIndex += step;

    gallerySlides.style.transition = "transform 0.6s ease";

    gallerySlides.style.transform = `translateX(${-galleryIndex * slideWidth}px)`;


    allowTransition = false;


    setTimeout(() => {

      allowTransition = true;

      const totalElements = gallerySlides.children.length;


      if(galleryIndex >= totalElements / 2){

        gallerySlides.style.transition = "none";

        galleryIndex = 0;

        gallerySlides.style.transform = `translateX(0px)`;

      }


      if(galleryIndex < 0){

        gallerySlides.style.transition = "none";

        galleryIndex = (totalElements / 2) - 1;

        gallerySlides.style.transform = `translateX(${-galleryIndex * slideWidth}px)`;

      }

    }, 650);

  }


  if(nextBtn) nextBtn.addEventListener("click", () => moveGallery(1));

  if(prevBtn) prevBtn.addEventListener("click", () => moveGallery(-1));


  setInterval(() => moveGallery(1), 4000);

}


// REVEAL

function revealSections(){

  const reveals = document.querySelectorAll(".reveal");

  const windowHeight = window.innerHeight;

  const revealPoint = 80;


  reveals.forEach(reveal => {

    const revealTop = reveal.getBoundingClientRect().top;


    if(revealTop < windowHeight - revealPoint){

      reveal.classList.add("visible");

    }

  });

}


window.addEventListener("scroll", revealSections);

window.addEventListener("load", revealSections);

document.addEventListener("DOMContentLoaded", revealSections);

setTimeout(revealSections, 300);

</script>
